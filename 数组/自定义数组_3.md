# 自定义数组_3

```java
import java.util.Arrays;

**实现动态扩容和缩容**

/**
 * 自定义一个Array
 * add时如果越界则扩容, remove时缩容
 */
public class Array<E> {
    // 真正用来存数据的地方
    // java中所有的基本数据类型都有包装类, java内部可以自动拆装箱操作
    private E[] data;

    // size的意义:
    // 1. 实际元素个数
    // 2. 指向第一个还没有赋值的索引
    private int size;

    // 构造函数
    public Array(int capacity) {
        data = (E[])new Object[capacity];
        size = 0;
    }

    public Array() {
        this(0);
    }

    // 获取数组的容量
    public int getCapacity() {
        return data.length;
    }

    // 获取数组中元素的个数
    public int getSize() {
        return size;
    }

    // 判断数组是否为空
    public boolean isEmpty() {
        return size == 0;
    }

    // 向数组的末尾添加元素
    public void addLast(E ele) {
        /**
        if (size == data.length) { // 说明数组已满, size此时位于数组末尾再往后面一格
            throw new RuntimeException("array is full!");
        }
        data[size++] = ele;
         */
        add(size, ele);
    }

    // 向数组起始位置添加元素
    public void addFirst(E ele) {
        add(0, ele);
    }

    // 往某个索引位置处添加元素
    public void add(int index, E ele) {
        if (size == data.length) { // 说明数组已满, size此时位于数组末尾再往后面一格
            resize(2*data.length); // 动态扩容为原来数据的2倍长
        }
        if (index < 0 || index > size) {
            throw new RuntimeException("index is not 0<index<size");
        }
        // 把i之后的元素都往后挪动
        for (int i = size-1; i>=index; i--) {
            data[i+1] = data[i];
        }
        data[index] = ele;
        size++;
    }

    // 扩容
    private void resize(int newCapacity) {
        E[] newData = (E [])new Object[newCapacity];
        for (int i = 0; i < size; i++) {
            newData[i] = data[i];
        }
        data = newData;
    }

    // 给定索引获得对应元素
    public E get(int index) {
        if (index < 0 || index > size) {
            throw new RuntimeException("index is illegal");
        }
        return data[index];
    }

    // 修改某个索引位置上的元素
    public void set(int index, E ele) {
        if (index < 0 || index > size) {
            throw new RuntimeException("index is illegal");
        }
        data[index] = ele;
    }

    // 查找是否包含指定元素
    public boolean contains(E ele) {
        for (int i = 0; i < size; i++) {
            if (data[i].equals(ele)) {
                return true;
            }
        }
        return false;
    }

    // 查找操作, 返回元素位置
    // 如果不存在指定元素, 则返回-1
    public int find(int ele) {
        for (int i = 0; i < size; i++) {
            if (data[i].equals(ele)) return i;
        }
        return -1;
    }

    // 删除指定索引的元素
    public E remove(int index) {
        if (index < 0 || index > size) {
            throw new RuntimeException("index is illegal");
        }
        E ret = data[index];
        // 前移index后面所有的元素
        for (int i = index + 1; i < size; i++) {
            data[i-1] = data[i];
        }
        size--;
        data[size] = null;

        // 判断是否需要缩容
        if (size == data.length/2) {
            resize(data.length/2);
        }

        return ret;
    }

    // 从数组中删除第一个元素, 返回删除的元素
    public E removeFirst() {
        return remove(0);
    }

    // 从数组中删除最后一个元素, 返回删除的元素
    public E removeLast() {
        return remove(size - 1);
    }

    // 从数组中删除元素
    public void removeElement(int ele) {
        int index = find(ele);
        if (index != -1) remove(index);
    }

    @Override
    public String toString() {
        return "Array{" +
                "data=" + Arrays.toString(data) +
                ", size=" + size +
                '}';
    }
}
```
